#include<stdio.h>


void towerofHanoi(int n,char src,char helper,char dest)
{
if(n==0)
return ;

towerofHanoi(n-1,src,helper,dest);
cout<<"Disc " << n << " :" << src << "-->" << dest << endl;
towerofHanoi(n-1,helper,dest,src);

}
